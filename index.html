async function submitData() {

  const photo = document.getElementById('photo').files[0];

  const clinicType = document.getElementById('clinicType').value;

  const clinicName = document.getElementById('clinicName').value;

  const clinicSize = document.getElementById('clinicSize').value;

  const status = document.getElementById('status');


  if (!photo || !clinicName) {

    alert("Enter clinic name and take a photo");

    return;

  }


  status.innerText = "Getting location...";


  navigator.geolocation.getCurrentPosition(position => {

    const reader = new FileReader();


    reader.onloadend = async () => {

      const base64Image = reader.result.split(',')[1];


      const payload = {

        latitude: position.coords.latitude,

        longitude: position.coords.longitude,

        image: base64Image,

        clinicType,

        clinicName,

        clinicSize

      };


      await fetch("https://script.google.com/macros/s/AKfycbzj1dwf7pPHYBbLBgyHZObxLNWw4UmCvEeerYrkTCpyIYc4oLcKtMz3wByMeSJ8pYmr7w/exec", {

        method: "POST",

        body: JSON.stringify(payload)

      });


      status.innerText = "Submitted ✅";

      document.getElementById('photo').value = "";

      document.getElementById('clinicName').value = "";

    };


    reader.readAsDataURL(photo);


  }, () => alert("Location permission denied"));

}

</script>
